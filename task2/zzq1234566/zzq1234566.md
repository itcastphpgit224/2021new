#include<stdio.h>
#include<stdlib.h>
struct ListNode{
	int val;
	struct ListNode *next;
};
int i;
/*将数组转化为链表*/ 
struct ListNode *Trans(int a[],int n){
	if(n == 0)
	{
		return NULL;
	}
	struct ListNode *ss= NULL; /*初始化*/ 
	struct ListNode *end= NULL;
	for(i=0;i<n;i++)
	{
		if(ss == NULL)
		{
			end=ss= (struct ListNode *)malloc(sizeof(struct ListNode));
			ss->val=a[i];
			ss->next=NULL;
			continue;
		}
		struct ListNode * news = (struct ListNode *)malloc(sizeof(struct ListNode));
		news->val=a[i];
		news->next=NULL;
		end->next=news;
		end = news;
	}
	return ss;
}
void Output(struct ListNode* head)
{
	printf("[");
	while(head!=NULL){
		if(head->next == NULL){
			printf("%d]",head->val);
			break;
		}
		printf("%d,",head->val);
		head = head->next;
	}
	printf("\n");
}
struct ListNode* addTwoNumbers(struct ListNode* l1, struct ListNode* l2){
    struct ListNode* temp;
    struct ListNode* prev1=NULL;
    struct ListNode* prev2=NULL;
    struct ListNode* current1=l1;
    struct ListNode* current2=l2;
    int carry=0;
    int len1=0,len2=0;
    while(current1){       //作用等同于while(NULL!=current1) 
        len1++;
        prev1=current1;
        current1=current1->next;
    }
    while(current2){
        len2++;
        prev2=current2;
        current2=current2->next;
    }
    if(len1==len2)
    ;
    else if(len1>len2){              //当2小于1时，为之申请内存 
        int nums=len1-len2;
        for (i=0;i<nums;i++){
            prev2->next=malloc(sizeof(struct ListNode));
            prev2->next->val=0;
            prev2=prev2->next;
        }
        prev2->next=NULL;
    }   
    else{                           //当1小于2时，为之申请内存 
        int nums=len2-len1;
        for (i=0;i<nums;i++){
            prev1->next=malloc(sizeof(struct ListNode));
            prev1->next->val=0;
            prev1=prev1->next;
        }
        prev1->next=NULL;
    }
    current1=l1;
    current2=l2;
    while(current1&&current2){      
        if (current1->val+current2->val+carry<10){
            current1->val=(current1->val+current2->val+carry)%10;
            carry=0;
        }
        else {             //处理进位 
            current1->val=(current1->val+current2->val+carry)%10;
            carry=1;
        }
        if (current1->next==NULL){
            temp=current1;
        }
        current1=current1->next;
        current2=current2->next;
    }
    if (carry){
        temp->next=malloc(sizeof(struct ListNode));
        temp->next->val=1;
        temp->next->next=NULL;
    }
    return l1;
}
int main()
{
	int a[]={2,4,5};

	struct ListNode *l1=Trans(a,3);
	struct ListNode *l2=Trans(b,3);
	addTwoNumbers(l1,l2);
	Output(l1);
